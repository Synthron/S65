ZP:
	$E0 HW_ROM Bank
	$E1 HW_RAM Bank
	$E2 HW_IOPAGE
	$E3 HW_SYSCONF (6502_mode, ROM_Pageout ...)
	$E4 free
	$E5 free
	$E6 free
	$E7 free


0000 7FFF		32k Fixed RAM
8000 BEFF		16k Paged RAM
		 BFxx 	Default IO Space
C000 FFFF		16k Paged ROM Space


RAMLO = /{IO} & /{A15} & /(ZPREG)
			= /(IO | A15 | ZPREG)
RAMHI = A15 & (/{A14} & /{IO} | PAGEOUT & /{IO})
			= A15 & (/{A14 | IO} | {PAGEOUT /{IO}})
ROM	  = A15 & A14 & /{PAGE_OUT} & /{IO}
